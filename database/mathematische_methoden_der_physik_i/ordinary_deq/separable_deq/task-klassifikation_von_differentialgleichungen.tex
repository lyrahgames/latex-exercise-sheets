\begin{atiTask}[
	title = Klassifikation von gewöhnlichen Differentialgleichungen,
	language = Deutsch,
]
	Klassifizieren Sie die folgenden gewöhnlichen Differentialgleichungen durch ihre Ordnung, Homogenität, Linearität, Separabilität.
	Sollte es sich um eine separable Differentialgleichung handeln, lösen Sie sie.
	% \begin{multicols}{2}
	\begin{atiSubequations}
		\begin{minipage}{0.5\textwidth}
		\item{
			\nfdrv[2]{y(x)}{x} + 2\fdrv{y(x)}{x} = y
		}
		\item{
			\fdrv{y(x)}{x} + \sin y - x^2 = 0
		}
		\item{
			y' + \tan(x)\cdot y = 0
		}
		\end{minipage}
		\begin{minipage}{0.5\textwidth}
		\item{
			\frac{y''}{y'} + x = 0
		}
		\item{
			yy'-x = 0
		}
		\item{
			\frac{x+1}{y+2} = \fdrv{y(x)}{x}
		}
		\end{minipage}
		\item{
			\sqrt{y^2 + 3a^2 + ya\curvb{2-\frac{4a}{2y}}} + \sqrt{x^2-a^2}\fdrv{y(x)}{x} = \sqrt{x+a}
		}
	\end{atiSubequations}
	% \end{multicols}
\end{atiTask}
