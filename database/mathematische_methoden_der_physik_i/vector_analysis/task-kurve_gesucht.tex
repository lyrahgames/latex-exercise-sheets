\begin{atiTask}[
  title = Parametrisierung gegeben, Kurve gesucht
]
  Skizzieren Sie die Kurven $xy$-Ebene, die durch die folgenden Parametrisierungen gegeben werden.
  \begin{atiSubequations}
    \item{
      \function{r}{[-1,1]}{\setR^3}
      \separate
      r(s) \define \frac{1}{2}\boxb{(1-s)\vec{i} + (-7+3s)\vec{j}}
    }
    \item{
      \function{r}{\boxb{\frac{\pi}{2},\frac{\pi}{2}}}{\setR^3}
      \separate
      r(t) \define 2\cos t \vec{i} + 2\sin t \vec{j}
    }
    \item{
      \function{r}{[0,2\pi]}{\setR^3}
      \separate
      r(t) \define t\vec{i} + \curvb{\frac{1}{2}\cos t + \frac{3}{2}}\vec{j}
    }
    \item{
      \function{r}{[1,2\pi]}{\setR^3}
      \separate
      r(t) \define t\vec{i} + \frac{7t^2 - 2 - 20\pi^2}{4\pi^2 - 1}\vec{j}
    }
  \end{atiSubequations}
\end{atiTask}
